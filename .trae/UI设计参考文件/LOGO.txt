    <script>
        // 主题切换逻辑
        const themeToggleBtn = document.getElementById('theme-toggle');
        const themeToggleDarkIcon = document.getElementById('theme-toggle-dark-icon');
        const themeToggleLightIcon = document.getElementById('theme-toggle-light-icon');

        if (document.documentElement.classList.contains('dark')) {
            themeToggleLightIcon.classList.remove('hidden');
        } else {
            themeToggleDarkIcon.classList.remove('hidden');
        }

        themeToggleBtn.addEventListener('click', function() {
            themeToggleDarkIcon.classList.toggle('hidden');
            themeToggleLightIcon.classList.toggle('hidden');
            document.documentElement.classList.toggle('dark');
        });

        // 注入到剪贴板的 CSS 样式 (确保纯净独立)
        const injectedCSS = `
            <style>
                .svg-logo-primary { fill: #2563eb; }
                .svg-logo-secondary { fill: #0ea5e9; }
                .svg-logo-text { fill: #0f172a; }
                .svg-stroke-primary { stroke: #2563eb; stroke-width: inherit; fill: none; }
                .svg-stroke-secondary { stroke: #0ea5e9; stroke-width: inherit; fill: none; }
                .svg-stroke-text { stroke: #0f172a; stroke-width: inherit; fill: none; }
                
                @media (prefers-color-scheme: dark) {
                    .svg-logo-primary { fill: #38bdf8; }
                    .svg-logo-secondary { fill: #818cf8; }
                    .svg-logo-text { fill: #f8fafc; }
                    .svg-stroke-primary { stroke: #38bdf8; }
                    .svg-stroke-secondary { stroke: #818cf8; }
                    .svg-stroke-text { stroke: #f8fafc; }
                }
            </style>
        `;

        function copySmartSVG(containerId) {
            const container = document.getElementById(containerId);
            const svgElement = container.querySelector('svg');
            
            const serializer = new XMLSerializer();
            let svgString = serializer.serializeToString(svgElement);
            
            // 插入内联 CSS 使得复制出的 SVG 具有独立的主题响应能力
            svgString = svgString.replace(/(<svg[^>]*>)/i, '$1' + injectedCSS);
            
            const textarea = document.createElement('textarea');
            textarea.value = svgString;
            document.body.appendChild(textarea);
            textarea.select();
            
            try {
                document.execCommand('copy');
                showToast();
            } catch (err) {
                alert('复制失败，请手动选择代码复制。');
            }
            
            document.body.removeChild(textarea);
        }

        function showToast() {
            const toast = document.getElementById("toast");
            toast.className = "toast show";
            setTimeout(function(){ toast.className = toast.className.replace("show", ""); }, 3000);
        }
    </script>
